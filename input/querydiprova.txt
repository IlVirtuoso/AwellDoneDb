CREATE TABLE site(username TEXT NOT NULL, id INT NOT NULL AUTO_INCREMENT, password TEXT, PRIMARY KEY(id,username));
CREATE TABLE applicazioni(id INT NOT NULL, name TEXT NOT NULL, PRIMARY KEY(name), FOREIGN KEY(id) REFERENCES site(id));
CREATE TABLE operatori(id INT NOT NULL, username TEXT NOT NULL, PRIMARY KEY(id,username), FOREIGN KEY(id,username) REFERENCES site(id,username));
